Rank-Based Command Dispatcher

Moderation Rank-Based Config-Driven

A chat command system with rank-checked permissions and a shared argument-validation pattern, built so new commands can be added without repeating permission or parsing logic.

DispatcherConfig.lua 
local Commands = {}

Commands.RequiredArgs = {
  kick = { "player", "reason" },
  ban  = { "player", "duration", "reason" },
  warn = { "player", "reason" },
}

Commands.RankRequired = {
  kick = 2,
  ban  = 3,
  warn = 1,
}

local function hasPermission(rank, command)
  return rank >= (Commands.RankRequired[command] or math.huge)
end

local function validateArgs(command, args)
  for _, argName in Commands.RequiredArgs[command] or {} do
    if args[argName] == nil then
      return false, "Missing arg: " .. argName
    end
  end
  return true
end

return Commands
